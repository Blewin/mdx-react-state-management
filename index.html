class MyComponentWithState extends Component {
  state = {
    text: 'some value',
    foo: 'another value',
    array: [1, 2, 3],
    object: {
      a: true,
      b: false,
      c: 5,
    },
  }

  render() {
    return (
      <div>
        <div>Current value:{this.state.text}</div>

        <button onClick={this._onClick}>Click me</button>
      </div>
    )
  }

  _onClick = () => {
    this.setState({
      text: 'the button has been clicked',
      foo: '8',
    })
  }
}

React Context to the rescue

How to avoid prop drilling

and pass data in the background

const MyContext = React.createContext()
// MyContext is an object: { Consumer, Provider }

class App extends Component {
  state = {
    balances: [0],
  }

  render() {
    return (
      <MyContext.Provider value={this.state.balances}>
        Other components here
      </MyContext.Provider>
    )
  }
}

class ContextUser extends Component {
  render() {
    return (
      <MyContext.Consumer>
        {balances => (
          <div>
            Balances can be used here.
            For example:
            There are {balances.length} accounts.
          </div>
        )}
      </MyContext.Consumer>
    )
  }
}
